   67   typedef struct
   68   {
   69       unsigned char isValid;       /* zero if the file has no audio */
   70   
   71       /* The following fields refer to the PCM format of the audio data as encoded in the file/device */
   72       unsigned char isFloat;       /* true if the samples are in floating-point format */
   73       double sampleRate;           /* audio sample rate */
   74       int numBits;                 /* audio bit depth */
   75       int numChannels;             /* number of audio channels (typically 1 or two) */
   76   
   77       /* The following fields refer to the format of the data in MATLAB/Simulink */
   78       AudioDataType dataType;      /* data type of audio samples */
   79       int frameSize;               /* number of audio samples per frame */
   80   
   81       /* The name of the compression format, if we are writing a file */
   82       const char* audioCompressor; /* set to NULL for none */
   83   } MMAudioInfo;
