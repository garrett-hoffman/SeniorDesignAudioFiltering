    9   void createAudioInfo(void *audioInfo, unsigned char isValid,
   10                        unsigned char isFloat,
   11                        double sampleRate, int numBits,
   12                        int numChannels, int frameSize,
   13                        int dataType, void* audioCompressor)
   14   {
   15       MMAudioInfo *pAudioInfo = (MMAudioInfo *) audioInfo;
   16       pAudioInfo->isValid         = isValid;
   17       pAudioInfo->isFloat         = isFloat;
   18       pAudioInfo->sampleRate      = sampleRate;
   19       pAudioInfo->numBits         = numBits;
   20       pAudioInfo->numChannels     = numChannels;
   21       pAudioInfo->frameSize       = frameSize;
   22       pAudioInfo->dataType        = (AudioDataType) dataType;
   23       pAudioInfo->audioCompressor = (char *)audioCompressor;
   24   }

   26   void createVideoInfo(void *videoInfo, unsigned char isValid,
   27                        double frameRate, double frameRateComputed,
   28                        const char *fourcc, int numPorts, int numBands,
   29                        int bandWidth, int bandHeight, unsigned char useMMReader,
   30                        int dataType, int orientation,
   31                        void* videoCompressor)
   32   {
   33       int i=0;
   34       MMVideoInfo *pVideoInfo = (MMVideoInfo *) videoInfo;
   35       pVideoInfo->isValid           = isValid;
   36       pVideoInfo->frameRate         = frameRate;
   37       pVideoInfo->frameRateComputed = frameRateComputed;
   38       for(i=0; i<4; i++)
   39           pVideoInfo->fourcc[i] = fourcc[i];
   40       pVideoInfo->numPorts = numPorts;
   41       pVideoInfo->numBands = numBands;
   42       pVideoInfo->bandWidth[0]  = bandWidth;
   43       pVideoInfo->bandHeight[0] = bandHeight;
   44       if(fourcc[0] == 'Y') { /* "YUY2" */
   45           pVideoInfo->bandWidth[1]  = bandWidth/2;
   46           pVideoInfo->bandWidth[2]  = bandWidth/2;
   47           pVideoInfo->bandHeight[1] = bandHeight;
   48           pVideoInfo->bandHeight[2] = bandHeight;
   49       }
   50       else { /* " RGB" */
   51           pVideoInfo->bandWidth[1]  = bandWidth;
   52           pVideoInfo->bandWidth[2]  = bandWidth;
   53           pVideoInfo->bandHeight[1] = bandHeight;
   54           pVideoInfo->bandHeight[2] = bandHeight;
   55       }
   56       pVideoInfo->useMMReader = useMMReader;
   57       pVideoInfo->dataType          = (VideoDataType) dataType;
   58       pVideoInfo->orientation       = (VideoFrameOrientation) orientation;
   59       pVideoInfo->videoCompressor   = (char *)videoCompressor;
   60   }
